<template>
  <div class="form-container">
    <h2>Data Entry</h2>
    <form @submit.prevent="addData">
      <input v-model="formData.name" placeholder="Name" required />
      <input v-model="formData.age" type="number" placeholder="Age" required />
      <input v-model="formData.email" type="email" placeholder="Email" required />
      <input v-model="formData.city" placeholder="City" required />
      <input v-model="formData.country" placeholder="Country" required />
      <button type="submit">Add Entry</button>
    </form>
  </div>
</template>

<script>
export default {
  emits: ["add-entry"],
  data() {
    return {
      formData: { name: "", age: "", email: "", city: "", country: "" },
    };
  },
  methods: {
    addData() {
      this.$emit("add-entry", { ...this.formData });
      this.formData = { name: "", age: "", email: "", city: "", country: "" };
    },
  },
};
</script>

<style scoped>
.form-container {
  padding: 1rem;
  border-bottom: 2px solid #ccc;
}
form {
  display: flex;
  flex-wrap: wrap;
  gap: 10px;
}
input {
  padding: 8px;
  border: 1px solid #ccc;
}
button {
  padding: 8px;
  background: blue;
  color: white;
  cursor: pointer;
}
</style>






<template>
  <div class="table-container">
    <h2>Data Table</h2>
    <ag-grid-vue
      class="ag-theme-alpine"
      style="height: 400px; width: 100%"
      :columnDefs="columnDefs"
      :rowData="rowData"
      :pagination="true"
      :paginationPageSize="5"
      :defaultColDef="defaultColDef"
    ></ag-grid-vue>
  </div>
</template>

<script>
import { AgGridVue } from "ag-grid-vue3";

export default {
  components: { AgGridVue },
  props: ["rowData"],
  data() {
    return {
      columnDefs: [
        { field: "name", sortable: true, filter: true },
        { field: "age", sortable: true, filter: "agNumberColumnFilter" },
        { field: "email", sortable: true, filter: true },
        { field: "city", sortable: true, filter: true },
        { field: "country", sortable: true, filter: true },
      ],
      defaultColDef: { resizable: true, flex: 1 },
    };
  },
};
</script>

<style>
.table-container {
  padding: 1rem;
}
.ag-theme-alpine {
  height: 400px;
}
</style>





<template>
  <div>
    <DataEntryForm @add-entry="addEntry" />
    <DataTable :rowData="rowData" />
  </div>
</template>

<script>
import DataEntryForm from "./components/DataEntryForm.vue";
import DataTable from "./components/DataTable.vue";

export default {
  components: { DataEntryForm, DataTable },
  data() {
    return {
      rowData: [
        { name: "Alice", age: 25, email: "alice@example.com", city: "New York", country: "USA" },
        { name: "Bob", age: 30, email: "bob@example.com", city: "London", country: "UK" },
        { name: "Charlie", age: 28, email: "charlie@example.com", city: "Berlin", country: "Germany" },
        { name: "David", age: 35, email: "david@example.com", city: "Paris", country: "France" },
        { name: "Eve", age: 22, email: "eve@example.com", city: "Madrid", country: "Spain" },
      ],
    };
  },
  methods: {
    addEntry(newEntry) {
      this.rowData.push(newEntry);
    },
  },
};
</script>









import { createApp } from "vue";
import App from "./App.vue";
import "ag-grid-community/styles/ag-grid.css";
import "ag-grid-community/styles/ag-theme-alpine.css";

createApp(App).mount("#app");
